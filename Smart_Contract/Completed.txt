pragma solidity ^0.8.0;

contract RentalContract {

    // ─────────────────────────────────────────────
    // BIẾN TRẠNG THÁI
    // ─────────────────────────────────────────────
    address public immutable owner;           // Ví của người cho thuê
    address public immutable renter;          // Ví của người thuê
    address public immutable arbitrator;      // Admin platform — phân xử tranh chấp

    uint256 public immutable rentalAmount;    // Tiền thuê (chuyển cho Owner sau khi Renter xác nhận nhận đồ)
    uint256 public depositAmount;   // Tiền cọc (giữ trong contract đến khi hoàn tất)
    uint256 public immutable lateFeePerDay;   // Phí phạt mỗi ngày trễ hạn
    uint256 public immutable itemValue;       // Giá trị món đồ — dùng để tính trần đền bù hư hại
    uint256 public immutable maxDamagePercent; // Tỷ lệ đền bù tối đa 

    uint256 public rentalStartTime;           // Thời điểm hợp đồng chuyển sang ACTIVE
    uint256 public rentalEndTime;             // Hạn chót trả đồ
    uint256 public returnedTime;              // Thời điểm Renter bấm trả đồ
    uint256 public damageCharges;             // Tổng tiền khấu trừ (phí trễ + hư hại)

    string public renterReturnPhotoHash;      // Hash IPFS ảnh Renter chụp lúc trả đồ
    string public renterReturnVideoHash;

    // ─────────────────────────────────────────────
    // TRẠNG THÁI HỢP ĐỒNG
    // ─────────────────────────────────────────────
    enum RentalStatus {
        INITIALIZED,                   // Hợp đồng vừa được tạo, chờ Renter nạp tiền
        ACTIVE,                        // Renter đã nạp tiền, chờ giao đồ
        ITEM_RECEIVED,                 // Renter xác nhận nhận đồ → tiền thuê chuyển cho Owner
        RETURNED_PENDING_INSPECTION,   // Renter đã trả đồ kèm ảnh, chờ Owner kiểm tra
        DEDUCTION_PROPOSED,            // Owner đề xuất khấu trừ hư hại
        DISPUTED,                      // Một bên gọi tranh chấp, chờ Admin phân xử
        COMPLETED                      // Hợp đồng kết thúc, tiền đã được phân phối
    }

    RentalStatus public rentalStatus;


    // ─────────────────────────────────────────────
    // SỰ KIỆN (EVENTS)
    // Nhóm 2 lắng nghe để cập nhật UI realtime
    // ─────────────────────────────────────────────
    event RentalFunded(
        address indexed renter,
        uint256 rentalAmount,
        uint256 depositAmount,
        uint256 rentalEndTime,
        uint256 timestamp
    );

    // Renter xác nhận nhận đồ → tiền thuê chuyển cho Owner
    event ItemReceived(
        address indexed renter,
        uint256 rentalAmountPaidToOwner,
        uint256 timestamp
    );

    event ItemReturned(
        address indexed renter,
        string photoHash,
        string videoHash,
        uint256 returnedTime,
        uint256 lateFeesApplied,
        uint256 timestamp
    );

    event DamageDeductionProposed(
        address indexed owner,
        uint256 proposedDeduction,
        uint256 totalDamageCharges,
        uint256 maxAllowedDeduction,
        uint256 timestamp
    );

    event DeductionAcceptedByRenter(
        address indexed renter,
        uint256 agreedDamageCharges,
        uint256 timestamp
    );

    event DisputeRaised(
        address indexed raisedBy,
        string reason,
        uint256 timestamp
    );

    event DisputeResolved(
        address indexed arbitrator,
        uint256 ownerPayout,
        uint256 renterRefund,
        string resolution,
        uint256 timestamp
    );

    event RentalConfirmed(
        address indexed owner,
        uint256 ownerPayout,
        uint256 renterRefund,
        uint256 timestamp
    );

    event AutoSettled(
        address indexed settler,
        uint256 ownerPayout,
        uint256 renterRefund,
        string reason,
        uint256 timestamp
    );

    event StatusTransitioned(
        RentalStatus previousStatus,
        RentalStatus newStatus,
        uint256 timestamp
    );


    // ─────────────────────────────────────────────
    // KHỞI TẠO HỢP ĐỒNG
    // Được gọi bởi RentalFactory mỗi khi tạo đơn thuê mới
    // ─────────────────────────────────────────────
    constructor(
        address _renter,
        address _arbitrator,
        uint256 _rentalAmount,
        uint256 _lateFeePerDay,
        uint256 _itemValue,
        uint256 _maxDamagePercent
    ) {
        require(_renter != address(0), "Invalid renter address");
        require(_arbitrator != address(0), "Invalid arbitrator address");
        require(_renter != msg.sender, "Renter cannot be owner");
        require(_rentalAmount > 0, "Rental amount must be positive");
        require(_lateFeePerDay > 0, "Late fee per day must be positive");
        require(_itemValue > 0, "Item value must be positive");
        require(_maxDamagePercent > 0 && _maxDamagePercent <= 100, "Max damage percent must be 1-100");

        owner = msg.sender;
        renter = _renter;
        arbitrator = _arbitrator;
        rentalAmount = _rentalAmount;
        lateFeePerDay = _lateFeePerDay;
        itemValue = _itemValue;
        maxDamagePercent = _maxDamagePercent;
        rentalStatus = RentalStatus.INITIALIZED;
    }


    // ─────────────────────────────────────────────
    // MODIFIER — Bộ lọc trạng thái và quyền
    // ─────────────────────────────────────────────
    modifier onlyInitialized() {
        require(rentalStatus == RentalStatus.INITIALIZED, "Must be INITIALIZED");
        _;
    }

    modifier onlyActive() {
        require(rentalStatus == RentalStatus.ACTIVE, "Must be ACTIVE");
        _;
    }

    modifier onlyItemReceived() {
        require(rentalStatus == RentalStatus.ITEM_RECEIVED, "Must be ITEM_RECEIVED");
        _;
    }

    modifier onlyReturnedPending() {
        require(rentalStatus == RentalStatus.RETURNED_PENDING_INSPECTION, "Must be RETURNED_PENDING_INSPECTION");
        _;
    }

    modifier onlyDeductionProposed() {
        require(rentalStatus == RentalStatus.DEDUCTION_PROPOSED, "Must be DEDUCTION_PROPOSED");
        _;
    }

    modifier onlyDisputed() {
        require(rentalStatus == RentalStatus.DISPUTED, "Must be DISPUTED");
        _;
    }

    modifier onlyRenter() {
        require(msg.sender == renter, "Only renter can call this");
        _;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this");
        _;
    }

    modifier onlyArbitrator() {
        require(msg.sender == arbitrator, "Only arbitrator can call this");
        _;
    }


    // ─────────────────────────────────────────────
    // BƯỚC 1: Renter nạp tiền → kích hoạt hợp đồng
    // Renter nạp đúng rentalAmount + depositAmount
    // Toàn bộ tiền giữ trong contract, chưa chuyển cho Owner
    // Tiền thuê chỉ chuyển sau khi Renter xác nhận nhận đồ (Bước 2)
    // ─────────────────────────────────────────────
    function calculateDepositAmount() public view returns (uint256) {
        return itemValue * 70 / 100;
}

    function fundRental(uint256 _rentalDays)
        external
        payable
        onlyInitialized
        onlyRenter
    {
        require(_rentalDays > 0, "Duration must be at least 1 day");
    uint256 computedDeposit = calculateDepositAmount();
    require(
        msg.value == rentalAmount + computedDeposit,
        "Payment must equal rentalAmount + depositAmount (70% of itemValue)"
    );

        depositAmount = computedDeposit; 
        rentalStartTime = block.timestamp;
        rentalEndTime = block.timestamp + (_rentalDays * 1 days);
        damageCharges = 0;

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.ACTIVE;

        // Chưa chuyển tiền thuê — giữ toàn bộ trong contract
        // đến khi Renter xác nhận nhận đồ

        emit RentalFunded(renter, rentalAmount, depositAmount, rentalEndTime, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);
    }


    // ─────────────────────────────────────────────
    // BƯỚC 2: Renter xác nhận đã nhận và kiểm tra đồ OK
    // → Tiền thuê được chuyển cho Owner ngay lúc này
    // Lý do: Renter đã nhận đồ thực tế → Owner xứng đáng nhận tiền thuê
    // Bảo vệ Renter: nếu đồ có vấn đề, Renter có thể báo trước
    //   thay vì bị đổ lỗi hư hại có sẵn lúc trả đồ sau này
    // ─────────────────────────────────────────────
    function confirmItemReceived()
        external
        onlyActive
        onlyRenter
    {
        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.ITEM_RECEIVED;

        emit ItemReceived(renter, rentalAmount, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);

        // Chuyển tiền thuê cho Owner sau khi Renter xác nhận nhận đồ
        (bool success, ) = payable(owner).call{value: rentalAmount}("");
        require(success, "Rental payment to owner failed");
    }


    // ─────────────────────────────────────────────
    // BƯỚC 3: Renter trả đồ + chụp ảnh bằng chứng
    // Bắt buộc truyền vào hash ảnh IPFS (Nhóm 3 upload lên IPFS trước)
    // Không có ảnh → không trả được → tránh tranh chấp mơ hồ

    // ─────────────────────────────────────────────
    function returnItem(
        string calldata _photoHash,
        string calldata _videoHash
)
        external
        onlyItemReceived
        onlyRenter
    {
        require(bytes(_photoHash).length > 0, "Photo proof is required before returning");
        require(bytes(_videoHash).length > 0, "Video proof is required before returning");

        returnedTime = block.timestamp;
        renterReturnPhotoHash = _photoHash;
        renterReturnVideoHash = _videoHash;

        uint256 lateFeesApplied = 0;
        if (block.timestamp > rentalEndTime) {
            // Ceiling division: làm tròn lên số ngày trễ
            // +1 days - 1 để đảm bảo trễ dù 1 giây cũng tính thêm 1 ngày
            uint256 overdueDays = (block.timestamp - rentalEndTime + 1 days - 1) / 1 days;
            lateFeesApplied = overdueDays * lateFeePerDay;

            // Phí trễ tối đa bằng tiền cọc, không vượt quá
            if (lateFeesApplied > depositAmount) {
                lateFeesApplied = depositAmount;
            }
            damageCharges = lateFeesApplied;
        }

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.RETURNED_PENDING_INSPECTION;

        emit ItemReturned(renter, _photoHash, _videoHash, returnedTime, lateFeesApplied, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);
    }


    // ─────────────────────────────────────────────
    // HÀM TIỆN ÍCH: Tính trần đền bù hư hại
    // Owner không được đề xuất vượt quá mức này
    // Công thức: itemValue * maxDamagePercent / 100
    // Ví dụ: itemValue = 2,000,000đ, maxDamagePercent = 70
    //   → maxAllowed = 1,400,000đ
    // Nếu trần vượt depositAmount thì lấy depositAmount làm giới hạn
    // ─────────────────────────────────────────────
    function getMaxDamageDeduction() public view returns (uint256) {
        uint256 maxByPolicy = itemValue * maxDamagePercent / 100;
        // Không thể đền bù quá số tiền cọc đang giữ (trừ phí trễ đã tính)
        uint256 remainingDeposit = depositAmount - damageCharges;
        return maxByPolicy < remainingDeposit ? maxByPolicy : remainingDeposit;
    }


    // ─────────────────────────────────────────────
    // BƯỚC 4A: Owner xác nhận nhận đồ, không hư hại
    // Contract phân phối tiền cọc:
    //   → Owner nhận phần phí trễ (nếu có)
    //   → Renter nhận lại phần còn lại
    // ─────────────────────────────────────────────
    function confirmReceipt()
        external
        onlyReturnedPending
        onlyOwner
    {
        uint256 ownerAdditionalPayout = damageCharges;  // Chỉ phí trễ, không có hư hại
        uint256 renterRefund = depositAmount - damageCharges;

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.COMPLETED;

        emit RentalConfirmed(owner, ownerAdditionalPayout, renterRefund, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);

        if (ownerAdditionalPayout > 0) {
            (bool ownerSuccess, ) = payable(owner).call{value: ownerAdditionalPayout}("");
            require(ownerSuccess, "Payment to owner failed");
        }
        (bool renterSuccess, ) = payable(renter).call{value: renterRefund}("");
        require(renterSuccess, "Refund to renter failed");
    }


    // ─────────────────────────────────────────────
    // BƯỚC 4B: Owner đề xuất khấu trừ hư hại bổ sung
    // Giới hạn bởi getMaxDamageDeduction():
    //   tối đa = itemValue * maxDamagePercent / 100
    //   và không vượt quá phần cọc còn lại sau phí trễ
    // Tiền đền bù hư hại được trừ vào tiền cọc của Renter
    // ─────────────────────────────────────────────
    function proposeDamageDeduction(uint256 _additionalDeduction)
        external
        onlyReturnedPending
        onlyOwner
    {
        require(_additionalDeduction > 0, "Deduction must be greater than 0");

        uint256 maxAllowed = getMaxDamageDeduction();
        require(
            _additionalDeduction <= maxAllowed,
            "Deduction exceeds maximum allowed (itemValue * maxDamagePercent / 100)"
        );

        damageCharges += _additionalDeduction;

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.DEDUCTION_PROPOSED;

        emit DamageDeductionProposed(owner, _additionalDeduction, damageCharges, maxAllowed, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);
    }


    // ─────────────────────────────────────────────
    // BƯỚC 5A: Renter đồng ý với mức khấu trừ
    // damageCharges được trừ vào tiền cọc:
    //   → Owner nhận damageCharges (phí trễ + đền bù hư hại)
    //   → Renter nhận lại depositAmount - damageCharges
    // ─────────────────────────────────────────────
    function acceptDeduction()
        external
        onlyDeductionProposed
        onlyRenter
    {
        uint256 ownerAdditionalPayout = damageCharges;
        uint256 renterRefund = depositAmount - damageCharges;

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.COMPLETED;

        emit DeductionAcceptedByRenter(renter, damageCharges, block.timestamp);
        emit RentalConfirmed(owner, ownerAdditionalPayout, renterRefund, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);

        if (ownerAdditionalPayout > 0) {
            (bool ownerSuccess, ) = payable(owner).call{value: ownerAdditionalPayout}("");
            require(ownerSuccess, "Payment to owner failed");
        }
        (bool renterSuccess, ) = payable(renter).call{value: renterRefund}("");
        require(renterSuccess, "Refund to renter failed");
    }


    // ─────────────────────────────────────────────
    // BƯỚC 5B: Gọi tranh chấp → Admin phân xử
    // Cả Owner và Renter đều có thể gọi
    // Được phép ở RETURNED_PENDING_INSPECTION hoặc DEDUCTION_PROPOSED
    // Khi DISPUTED: autoSettle không còn hiệu lực, phải chờ Admin
    // ─────────────────────────────────────────────
    function raiseDispute(string calldata _reason)
        external
    {
        require(
            msg.sender == owner || msg.sender == renter,
            "Only owner or renter can raise dispute"
        );
        require(
            rentalStatus == RentalStatus.RETURNED_PENDING_INSPECTION ||
            rentalStatus == RentalStatus.DEDUCTION_PROPOSED,
            "Dispute only allowed in RETURNED_PENDING_INSPECTION or DEDUCTION_PROPOSED"
        );
        require(bytes(_reason).length > 0, "Reason is required");

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.DISPUTED;

        emit DisputeRaised(msg.sender, _reason, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);
    }


    // ─────────────────────────────────────────────
    // BƯỚC 6: Admin phân xử tranh chấp
    // Admin xem ảnh IPFS của Renter + bằng chứng 2 bên
    // → quyết định _finalDeduction hợp lý
    // → contract tự động thực thi
    // Renter không thể từ chối vì tiền cọc đã bị khóa trong contract
    // Admin cũng bị giới hạn bởi maxDamagePercent để tránh lạm quyền
    // ─────────────────────────────────────────────
    function resolveDispute(uint256 _finalDeduction, string calldata _resolution)
        external
        onlyDisputed
        onlyArbitrator
    {
        require(_finalDeduction <= depositAmount, "Deduction cannot exceed deposit");
        require(bytes(_resolution).length > 0, "Resolution note is required");

        uint256 ownerPayout = _finalDeduction;
        uint256 renterRefund = depositAmount - _finalDeduction;

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.COMPLETED;

        emit DisputeResolved(arbitrator, ownerPayout, renterRefund, _resolution, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);

        if (ownerPayout > 0) {
            (bool ownerSuccess, ) = payable(owner).call{value: ownerPayout}("");
            require(ownerSuccess, "Payment to owner failed");
        }
        (bool renterSuccess, ) = payable(renter).call{value: renterRefund}("");
        require(renterSuccess, "Refund to renter failed");
    }


    // ─────────────────────────────────────────────
    // AUTO SETTLE: Tự động tất toán sau 7 ngày im lặng
    // Áp dụng cho RETURNED_PENDING_INSPECTION và DEDUCTION_PROPOSED
    // KHÔNG áp dụng cho DISPUTED (phải chờ Admin phân xử)
    // Ai cũng có thể gọi sau khi hết 7 ngày
    // ─────────────────────────────────────────────
    function autoSettle() external {
        require(
            rentalStatus == RentalStatus.RETURNED_PENDING_INSPECTION ||
            rentalStatus == RentalStatus.DEDUCTION_PROPOSED,
            "Auto settle only for RETURNED_PENDING_INSPECTION or DEDUCTION_PROPOSED"
        );
        require(
            block.timestamp >= returnedTime + 7 days,
            "7-day inspection period not yet expired"
        );

        uint256 ownerAdditionalPayout = damageCharges;
        uint256 renterRefund = depositAmount - damageCharges;

        string memory reason;
        if (rentalStatus == RentalStatus.RETURNED_PENDING_INSPECTION) {
            reason = "Owner_Did_Not_Respond_Within_7_Days";
        } else {
            reason = "Renter_Did_Not_Accept_Damage_Proposal_Within_7_Days";
        }

        RentalStatus previousStatus = rentalStatus;
        rentalStatus = RentalStatus.COMPLETED;

        emit AutoSettled(msg.sender, ownerAdditionalPayout, renterRefund, reason, block.timestamp);
        emit StatusTransitioned(previousStatus, rentalStatus, block.timestamp);

        if (ownerAdditionalPayout > 0) {
            (bool ownerSuccess, ) = payable(owner).call{value: ownerAdditionalPayout}("");
            require(ownerSuccess, "Payment to owner failed");
        }
        (bool renterSuccess, ) = payable(renter).call{value: renterRefund}("");
        require(renterSuccess, "Refund to renter failed");
    }


    // ─────────────────────────────────────────────
    // HÀM TRUY VẤN (VIEW) — Nhóm 2 dùng để hiển thị UI
    // ─────────────────────────────────────────────

    // Trạng thái hợp đồng dạng chữ
    function getStatusString() external view returns (string memory) {
        if (rentalStatus == RentalStatus.INITIALIZED)                 return "INITIALIZED";
        if (rentalStatus == RentalStatus.ACTIVE)                      return "ACTIVE";
        if (rentalStatus == RentalStatus.ITEM_RECEIVED)               return "ITEM_RECEIVED";
        if (rentalStatus == RentalStatus.RETURNED_PENDING_INSPECTION) return "RETURNED_PENDING_INSPECTION";
        if (rentalStatus == RentalStatus.DEDUCTION_PROPOSED)          return "DEDUCTION_PROPOSED";
        if (rentalStatus == RentalStatus.DISPUTED)                    return "DISPUTED";
        return "COMPLETED";
    }

    // Số ngày trễ hạn hiện tại (ceiling division)
    function calculateOverdueDays() external view returns (uint256) {
        if (block.timestamp <= rentalEndTime) return 0;
        return (block.timestamp - rentalEndTime + 1 days - 1) / 1 days;
    }

    // Phí trễ hạn dự kiến chưa bị giới hạn bởi trần tiền cọc
    function calculatePotentialLateFee() external view returns (uint256) {
        if (block.timestamp <= rentalEndTime) return 0;
        uint256 overdueDays = (block.timestamp - rentalEndTime + 1 days - 1) / 1 days;
        return overdueDays * lateFeePerDay;
    }

    // Giây còn lại của 7 ngày tất toán — dùng cho đồng hồ đếm ngược trên UI
    // Trả về -1 nếu Renter chưa trả đồ, 0 nếu đã hết hạn
    function getRemainingSettlementPeriod() external view returns (int256) {
        if (returnedTime == 0) return -1;
        uint256 inspectionDeadline = returnedTime + 7 days;
        if (block.timestamp >= inspectionDeadline) return 0;
        return int256(inspectionDeadline - block.timestamp);
    }

// Hàm 1: thông tin cơ bản các bên + tài chính
function getContractInfo()
    external
    view
    returns (
        address _owner,
        address _renter,
        address _arbitrator,
        uint256 _rentalAmount,
        uint256 _depositAmount,
        uint256 _lateFeePerDay,
        uint256 _itemValue,
        uint256 _maxDamagePercent
    )
{
    return (
        owner, renter, arbitrator,
        rentalAmount, depositAmount, lateFeePerDay,
        itemValue, maxDamagePercent
    );
}

// Hàm 2: trạng thái thực thi + bằng chứng
function getContractState()
    external
    view
    returns (
        RentalStatus _status,
        uint256 _rentalStartTime,
        uint256 _rentalEndTime,
        uint256 _returnedTime,
        uint256 _damageCharges,
        string memory _renterReturnPhotoHash,
        string memory _renterReturnVideoHash
    )
{
    return (
        rentalStatus,
        rentalStartTime, rentalEndTime, returnedTime,
        damageCharges, renterReturnPhotoHash, renterReturnVideoHash
    );
}

    // Số tiền dự kiến mỗi bên nhận nếu tất toán ngay lúc này
    function getExpectedPayouts()
        external
        view
        returns (uint256 ownerAdditionalAmount, uint256 renterAmount)
    {
        if (
            rentalStatus != RentalStatus.RETURNED_PENDING_INSPECTION &&
            rentalStatus != RentalStatus.DEDUCTION_PROPOSED &&
            rentalStatus != RentalStatus.DISPUTED
        ) {
            return (0, 0);
        }
        return (damageCharges, depositAmount - damageCharges);
    }

    // Xem trước kết quả autoSettle + đồng hồ đếm ngược
    function getAutoSettlePreview()
        external
        view
        returns (
            uint256 ownerAdditionalPayout,
            uint256 renterRefund,
            uint256 secondsUntilAutoSettle,
            bool canAutoSettle
        )
    {
        if (returnedTime == 0) return (0, 0, 0, false);
        uint256 inspectionDeadline = returnedTime + 7 days;
        bool autoSettleReady = block.timestamp >= inspectionDeadline;
        uint256 secondsRemaining = autoSettleReady ? 0 : (inspectionDeadline - block.timestamp);
        return (
            damageCharges,
            depositAmount - damageCharges,
            secondsRemaining,
            autoSettleReady
        );
    }
}


// ─────────────────────────────────────────────────────────────────────────────
// FACTORY CONTRACT — Hợp đồng mẹ
// Tạo ra RentalContract riêng cho mỗi đơn thuê
// arbitrator được set 1 lần khi deploy Factory → áp dụng cho mọi contract con
// ─────────────────────────────────────────────────────────────────────────────
contract RentalFactory {

    address public immutable arbitrator;

    RentalContract[] public rentalContracts;
    mapping(address => RentalContract[]) public ownerRentals;
    mapping(address => RentalContract[]) public renterRentals;
    mapping(address => bool) public isValidRental;

    event RentalContractCreated(
        address indexed rentalAddress,
        address indexed itemOwner,
        address indexed itemRenter,
        uint256 rentalAmount,
        uint256 itemValue,
        uint256 maxDamagePercent
    );

    // arbitrator là admin platform, set khi deploy Factory
    constructor(address _arbitrator) {
        require(_arbitrator != address(0), "Invalid arbitrator address");
        arbitrator = _arbitrator;
    }

    // Owner gọi hàm này để tạo đơn thuê mới
    function createRentalContract(
        address _renter,
        uint256 _rentalAmount,
        uint256 _lateFeePerDay,
        uint256 _itemValue,
        uint256 _maxDamagePercent
    ) public returns (address) {
        require(_renter != address(0), "Invalid renter address");
        require(_renter != msg.sender, "Owner cannot rent to themselves");
        require(_rentalAmount > 0, "Rental amount must be positive");
        require(_lateFeePerDay > 0, "Late fee must be positive");
        require(_itemValue > 0, "Item value must be positive");
        require(_maxDamagePercent > 0 && _maxDamagePercent <= 100, "Max damage percent must be 1-100");

        // Deploy contract mới, truyền arbitrator từ Factory vào
        RentalContract newRental = new RentalContract(
            _renter,
            arbitrator,
            _rentalAmount,
            _lateFeePerDay,
            _itemValue,
            _maxDamagePercent
        );

        rentalContracts.push(newRental);
        ownerRentals[msg.sender].push(newRental);
        renterRentals[_renter].push(newRental);
        isValidRental[address(newRental)] = true;

        emit RentalContractCreated(
            address(newRental),
            msg.sender,
            _renter,
            _rentalAmount,
            _itemValue,
            _maxDamagePercent
        );

        return address(newRental);
    }

    function getTotalRentalContracts() public view returns (uint256) {
        return rentalContracts.length;
    }

    function getRentalContract(uint256 _index) public view returns (address) {
        require(_index < rentalContracts.length, "Index out of bounds");
        return address(rentalContracts[_index]);
    }

    function getOwnerRentals(address _owner) public view returns (RentalContract[] memory) {
        return ownerRentals[_owner];
    }

    function getOwnerRentalCount(address _owner) public view returns (uint256) {
        return ownerRentals[_owner].length;
    }

    function getRenterRentals(address _renter) public view returns (RentalContract[] memory) {
        return renterRentals[_renter];
    }

    function getAllRentalContracts() public view returns (RentalContract[] memory) {
        return rentalContracts;
    }
}
